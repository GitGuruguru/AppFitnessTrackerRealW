csharppublic class SegmentInfo
{
    public Color Color { get; set; }
    public float Degrees { get; set; } // e.g. 120f for 1/3 of circle

2. The drawable (draws only the arc strokes):
csharppublic class SegmentedCircleDrawable : IDrawable
{
    public List<SegmentInfo> Segments { get; set; } = new();
    public float StrokeWidth { get; set; } = 8f;
    public float GapDegrees { get; set; } = 3f; // gap between segments

    public void Draw(ICanvas canvas, RectF rect)
    {
        float cx = rect.Center.X;
        float cy = rect.Center.Y;
        float r = Math.Min(rect.Width, rect.Height) / 2 - StrokeWidth;
        float startAngle = -90f; // start at top

        foreach (var seg in Segments)
        {
            canvas.StrokeColor = seg.Color;
            canvas.StrokeSize = StrokeWidth;
            canvas.StrokeLineCap = LineCap.Butt; // or Round for rounded ends

            float sweep = seg.Degrees - GapDegrees;
            canvas.DrawArc(cx - r, cy - r, r * 2, r * 2,
                startAngle, sweep, clockwise: false, closed: false);

            startAngle += seg.Degrees;
        }
    }
}
